STATEMENT           :rule-def: DECLARATION ;
                    :end:

DECLARATION         :rule-def:
                        TYPE SYMBOL = EXPRESSION
                    :rule-def:
                        let TYPE SYMBOL be EXPRESSION
                    :end:

TYPE                :rule-def: int
                    :rule-def: double
                    :rule-def: boolean
                    :rule-def: SYMBOL
                    :end:

EXPRESSION          :rule-def: EXPRESSION-LITERAL OPERATOR EXPRESSION
                    :rule-def: \( EXPRESSION \)
                    :rule-def: EXPRESSION-LITERAL
                    :end:

EXPRESSION-LITERAL  :rule-def: SYMBOL
                    :rule-def: NUMBER
                    :end:

BOOLEAN-EXPRESSION  :rule-def: is BOOLEAN-EXPRESSION ?
                    :rule-def: NUMBER BOOLEAN-OPERATOR
                    :rule-def: SYMBOL
                    :end:

OPERATOR            :rule-def: MATH-OPERATOR
                    :rule-def: BOOLEAN-OPERATOR
                    :end:
MATH-OPERATOR       :rule-def: \+
                    :rule-def: -
                    :rule-def: \*
                    :rule-def: /
                    :end:
COMPARISON-OPERATOR :rule-def: >
                    :rule-def: <
                    :rule-def: !
                    :rule-def: =
                    :rule-def: = =
                    :end:
BOOLEAN-OPERATOR    :rule-def: and
                    :rule-def: or
                    :rule-def: not
                    :end:

STRING              :rule-def: " STRING-LITERALS " :end:
STRING-LITERALS     :rule-def: .+ STRING-LITERALS
                    :rule-def: .+
                    :end:

LITERAL             :rule-def: NUMBER
                    :rule-def: SYMBOL
                    :end:
NUMBER              :rule-def: DOUBLE-LITERAL
                    :rule-def: INTEGER-LITERAL
                    :end:
DOUBLE-LITERAL      :rule-def: \d+\.\d+             :end:
INTEGER-LITERAL     :rule-def: \d+                  :end:
SYMBOL              :rule-def: [a-zA-z]+            :end:
